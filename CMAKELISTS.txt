cmake_minimum_required(VERSION 3.17)

project(Squares VERSION 1.0)

set(CMAKE_CXX_STANDARD 17)

set (SDL_INCLUDE_DIR ${CMAKE_SOURCE_DIR}/vendor/sdl/SDL2-2.0.12/include)
set (SDL_LIB_DIR ${CMAKE_SOURCE_DIR}/vendor/sdl/SDL2-2.0.12/lib/x64)


add_executable(
    Squares
    "src/Squares.cpp"
    )

target_include_directories(Squares PRIVATE ${SDL_INCLUDE_DIR})
target_link_libraries( Squares ${SDL_LIB_DIR}/SDL2main.lib ${SDL_LIB_DIR}/SDL2.lib)
